\begin{exo}
On considère l'équation
\begin{equation}\label{eq:1}
47a-111b=2,\quad\textrm{avec }(a,b)\in\Z^2.\tag{$\star$}
\end{equation}
\begin{enumerate}
\item Donner un couple $(u,v)\in\Z^2$ vérifiant
\begin{equation*}
47u-111v=1.
\end{equation*}
\item Résoudre l'équation (\ref{eq:1}).
\item En utilisant la première question, montrer que $\overline{47}$ est
inversible dans l'anneau $\Z/111\Z$ et déterminer son inverse.
\end{enumerate}
\end{exo}

\begin{exo}
On considère un nombre premier $p$.
\begin{enumerate}
\item Si $k$ est un entier compris entre $1$ et $p-1$, montrer que $p$ divise le
coefficient binomial $\textrm{C}_p^k=\frac{p!}{k!(p-k)!}$.
\item En déduire par récurrence que pour tout $a$ dans $\N$, l'on a
\begin{equation*}
a^p\equiv a\pmod{p}.
\end{equation*}
\end{enumerate}
\end{exo}
