\begin{exo}
On considère l'équation
\begin{equation}\label{eq:1}
77a-26b=3,\quad\textrm{avec }(a,b)\in\Z^2.\tag{$\star$}
\end{equation}
\begin{enumerate}
\item Donner un couple $(u,v)\in\Z^2$ vérifiant
\begin{equation*}
77u-26v=1.
\end{equation*}
\item Résoudre l'équation (\ref{eq:1}).
\item En utilisant la première question, montrer que $\overline{26}$ est
inversible dans l'anneau $\Z/77\Z$ et déterminer son inverse.
\end{enumerate}
\end{exo}

\begin{exo}
On considère la suite $(F_n)_{n\ge 0}$ définie par
\begin{equation*}
F_0=F_1=1\quad\textrm{et}\quad\forall n\ge 1,\quad F_{n+1}=F_n+F_{n-1}.
\end{equation*}
\begin{enumerate}
\item Calculer $F_0,\dots,F_{10}$.
\item Montrer que si $a$ et $b$ sont deux entiers $\ge 1$ premiers entre eux, alors $a$ et $a+b$ le
sont aussi.
\item En déduire par récurrence que pour tout entier $n\ge 0$, $F_n$ et
$F_{n+1}$ sont premiers entre eux.
\end{enumerate}
\end{exo}
