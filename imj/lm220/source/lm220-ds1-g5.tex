\begin{exo}
On considère l'équation
\begin{equation}\label{eq:1}
26a+21b=2,\quad\textrm{avec }(a,b)\in\Z^2.\tag{$\star$}
\end{equation}
\begin{enumerate}
\item Donner un couple $(u,v)\in\Z^2$ vérifiant
\begin{equation*}
26u+21v=1.
\end{equation*}
\item Résoudre l'équation (\ref{eq:1}).
\item En utilisant la première question, montrer que $\overline{21}$ est
inversible dans l'anneau $\Z/26\Z$ et déterminer son inverse.
\end{enumerate}
\end{exo}

\begin{exo}
On considère deux entiers $a$ et $b$ $\ge 1$.
\begin{enumerate}
\item Montrer que si $a$ et $b$ sont premiers entre eux, alors $a$ et $a+b$ le
sont aussi.
\item Montrer que le pgcd de $2^a-1$ et $2^b-1$ est $2^d-1$ où $d$ est le pgcd
de $a$ et $b$.
\end{enumerate}
\end{exo}
