\begin{exo} ~\\
 \begin{enumerate}
  \item Expliquer pourquoi $12$ est inversible dans $\Z/25\Z$ et calculer son inverse.
  \item En déduire que l'équation $12x = 7$ admet $11$ pour \emph{unique} solution dans $\Z/25\Z$.
  \item Résoudre dans $\Z$ le système de congruences suivant.
  \[
   \left\{ \begin{aligned} 12x &\equiv 7 \modulo{25} \\ x &\equiv 2 \modulo{21} \end{aligned} \right.
  \]
 \end{enumerate}
\end{exo}

\bigskip

\begin{exo} ~\\
 \begin{enumerate}
  \item Donner la liste complète des éléments du groupe $(\Z/20\Z)^*$.
  \item Montrer que $(\Z/20\Z)^*$ n'est pas cyclique. \indic{On pourra au choix utiliser le théorème chinois, ou calculer l'ordre de chaque élément.}
  \item Calculer le reste de la division euclidienne de $383^{127}$ par $20$.
 \end{enumerate}
\end{exo}

\bigskip

\begin{qc} ~\\
 \begin{enumerate}
  \item Définir la fonction $\varphi$ d'Euler, et donner une formule explicite pour calculer $\varphi(n)$.
  \item Énoncer le théorème d'Euler.
 \end{enumerate}
\end{qc}
