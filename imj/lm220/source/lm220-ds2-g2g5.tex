\begin{exo} ~\\
 \begin{enumerate}
  \item Expliquer pourquoi $11$ est inversible dans $\Z/24\Z$ et calculer son inverse.
  \item En déduire que l'équation $11x = 3$ admet $9$ pour \emph{unique} solution dans $\Z/24\Z$.
  \item Résoudre dans $\Z$ le système de congruences suivant.
  \[
   \left\{ \begin{aligned} 11x & \equiv & 3 \modulo 24 \\ x & \equiv & 2 \modulo 29 \end{aligned} \right.
  \]
 \end{enumerate}
\end{exo}
