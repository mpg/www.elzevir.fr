\item Dans un cryptosystème utilisant la méthode RSA avec la clé publique $(n,e)=(91,5)$, on souhaite envoyer le message $M_1=9$. Déterminer le message codé $C_1$ à envoyer.

\begin{exo}
On considère les deux polynômes suivants de $\Q\left[X\right]$~:
\begin{equation*}
 P(X)=X^4+X^3+2X^2-X+3\quad\textrm{et}\quad Q(X)=X^3+1.
\end{equation*}
\begin{enumerate}
\item Calculer le pgcd de $P$ et $Q$.
\item En déduire la factorisation de $P$ en produit de facteurs irréductibles dans $\Q\left[X\right]$.
\end{enumerate}


\end{exo}

\begin{qc} ~\\
 Démontrer le théorème suivant.
\begin{thm}
 Soient $K$ un corps et $I$ un idéal non nul de $K\left[X\right]$. Il existe un unique polynôme unitaire ${P\in K\left[X\right]}$ tel que l'on ait $I=(P)$.
\end{thm}
\end{qc}
