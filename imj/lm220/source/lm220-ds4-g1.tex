\begin{exo}
 Parmi les anneaux suivants, dire lesquels sont des corps. Justifier soigneusement les réponses.
 \begin{enumerate}
  \item $\F_2[X]/ (X^2 + X + 1)$,
  \item $\F_3[X]/ (X^2 + X + 1)$,
  \item $\Q[X]/ (X^4 + 1)$,
  \item $\F_3[X]/ (X^4 + 1)$,
 \end{enumerate}
 Donner la caractéristique et le nombre d'éléments de chacun.
\end{exo}

\begin{exo}
 On considère l'anneau
 \[
  K = \F_3[X]/(X^3-X-1) \pmm.
 \]
 \begin{enumerate}
  \item Montrer que $K$ est un corps et donner sa caractéristique, ainsi que son nombre d'éléments.
  \item On note $\alpha$ la classe de $X$ dans $K$.
  \begin{enumerate}
   \item Quel est l'inverse de $1 + \alpha^2$ dans $K^\times$?
   \item Montrer que $\alpha^{12} = (1 + \alpha)^4$ et exprimer cet élément dans la base $(1, \alpha, \alpha^2)$.
   \item En déduire que $\alpha$ est d'ordre $13$ dans $K^\times$.
  \end{enumerate}
 \end{enumerate}
\end{exo}
