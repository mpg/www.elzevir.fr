\section{Généralités}

\begin{exo} Plans dans l'espace. \label{explan}
\begin{enumerate}
\item Dans $\R^3$, on considère l'ensemble $\mathcal{P} = \{(x, y, z) | x+3y+5z=0\}$. Montrer que c'est un sous-espace vectoriel de $\R^3$ et en donner une base. \label{ptplan}
\item On pose $v_1 = (-1, 0, 3)$ et $v_2 = (1, 1, 2)$. Montrer que c'est une famille libre, non génératrice. Déterminer une équation du sous-espace qu'elle engendre.
\end{enumerate}
\end{exo}

\begin{exo}
Dites si la famille $\mathcal{F}$ est libre, génératrice, et si elle forme une base de $E$ dans les cas suivants :
\begin{enumerate}
\item $\mathcal{F} = \left\{ \begin{pmatrix} 1 \\ 0 \\ 2 \end{pmatrix}, \begin{pmatrix} -1 \\ 3 \\ 2 \end{pmatrix}, \begin{pmatrix} 3 \\ -3 \\ 2 \end{pmatrix} \right\} $ et $E = \R^3$.
\item $\mathcal{F} = \{X^3-2X^2+X+3, 2X^3-2X^2+5X, -X^3+X^2+1 \}$ et\footnote{On rappelle que $\R_n[X]$ est l'espace des polynômes de degré au plus $n$.} $E = \R_3[X]$.
\item $\mathcal{F} = \left\{ \I[2], \begin{pmatrix} 1 & 1 \\ 0 & 1 \end{pmatrix}, \begin{pmatrix} -1 & 4 \\ -1 & 3 \end{pmatrix}, \begin{pmatrix} 2 & -4 \\ 3 & 1 \end{pmatrix} \right\}$ et $E = M_2(\R)$.
\end{enumerate}
\end{exo}

\begin{exo} Deux espaces de dimension infinie.
\begin{enumerate}
\item On considère $\mathcal{C} = \mathcal{C}^0([0; 1];\, \R)$ l'espace des fonctions continues de $[0; 1]$ dans $\R$. Montrer que l'application $l: \mathcal{C}\to\R,\ f \mapsto \int_0^1 f(t)\, dt$ est linéaire et surjective.
\item On considère $V$ le $\C$-espace vectoriel des suites à coefficients complexes, indexées par $\N$, muni de l'application $\Delta: V \to V,\ (u_n)_{n\ge0} \mapsto (u_{n+1})_{n\ge0}$. Montrer que c'est un endomorphisme; calculer son image et son noyau. Observer.
\end{enumerate}
\end{exo}

\section{Applications linéaires et matrices}

\begin{exo} On considère\footnotemark[1] $E = \R_3[X]$, muni de l'opération de dérivation $D : P \mapsto P'$.
\begin{enumerate}
\item Vérifier que D est linéaire, calculer son noyau et son image.
\item \'Enoncer le théorème du rang et le vérifier sur cet exemple.
\item \'Ecrire la matrice $M$ de $D$ dans la base canonique $(1, X, X^2, X^3)$.
\item On considère $D^2: E \to E,\ P \mapsto P''$; écrire sa matrice dans la même base.
\item Donner \emph{sans calculs} la valeur de $M^4$.
\end{enumerate}
\end{exo}

\begin{exo}
Soit $E$ un $\C$-espace vectoriel de dimension finie, et $f \in \mathcal{L}(E)$. Montrer que :
\[ E= \im f \oplus \ker f \Leftrightarrow \im f = \im f^2 \pmm{.}\]
Constater en particulier l'équivalence sur les cas suivants:
\begin{itemize}
\item $\Pi : \R^3 \to \R^3,\ (x, y, z) \mapsto (x, y, x-y)$,
\item $D : \R_3[X] \to \R_3[X],\ P \mapsto P'$.
\end{itemize}
\end{exo}

\begin{exo}
On considère l'application $f : \R^3 \to \R,\ (x, y, z) \mapsto x+3y+5z$. Montrer qu'elle est surjective. En déduire que l'ensemble de la question \ref{explan}.\ref{ptplan} est un sous-espace de dimension $2$.
\end{exo}

\begin{exo}
Soient $E$ un espace vectoriel et $f \in \mathcal{L}(E)$ telle que, pour tout $x \in E$, la famille $(x, f(x))$ est liée. Autrement dit, pour tout $x \in E$, il existe un scalaire, noté $\lambda_x$, tel que $f(x) = \lambda_x \cdot x$. Soient $x$ et $y$ deux vecteurs non nuls; montrer que $\lambda_x = \lambda_y$ en considérant les deux cas suivants :
\begin{itemize}
\item $x$ et $y$ sont liés,
\item $x$ et $y$ sont indépendants.
\end{itemize}
En déduire que $f$ est une homothétie.
\end{exo}

\begin{exo} Soient $E_{11} = \begin{pmatrix} 1 & 0 \\ 0 & 0 \end{pmatrix}$, $E_{12} = \begin{pmatrix} 0 & 1 \\ 0 & 0 \end{pmatrix}$, $E_{21} = \begin{pmatrix} 0 & 0 \\ 1 & 0 \end{pmatrix}$, $E_{22} = \begin{pmatrix} 0 & 0 \\ 0 & 1 \end{pmatrix}$; on rappelle $(E_{11}, E_{12}, E_{21}, E_{22})$ est une base, dite canonique, de $M_2(\C)$. On pose $A = \begin{pmatrix} 2&5\\1&3 \end{pmatrix}$ et on considère l'application $f : M_2(\C) \to M_2(\C),\ M \mapsto AM$.
\begin{enumerate}
\item Montrer que $f$ est linéaire et calculer sa matrice dans la base canonique.
\item Prouver que $f$ est inversible et calculer son inverse.
\end{enumerate}
\end{exo}

\section{Changement de base}

\begin{exo}
On considère $D: \R_3[X] \to \R_2[X],\ P \mapsto P'$ et $i: \R_2[X] \to \R_3[X],\ P \mapsto \int_0^X P(t)\,dt$. On pose par ailleurs $\mathcal{B}_2 = (1, 1-X, 1-2X+X^2)$ et $\mathcal{B}_3 = (1, 1-X, 1-2X+X^2, 1-3X+3X^2-X^3)$.
\begin{enumerate}
\item Les applications $D$ et $i$ sont-elles injectives, surjectives? Calculer $D\circ i$ et $i \circ D$, observer.
\item Justifier brièvement que $\mathcal{B}_2$ (resp. $\mathcal{B}_3$) est une base de $\R_2[X]$ (resp. $\R_3[X]$) et écrire les matrices de $D$ et de $i$ dans ces bases.
\item Calculer les matrices de passages de ces bases aux bases canoniques, en déduire par les formules de changement de base les matrices de $D$ et $i$ dans les bases canoniques. Vérifier ce dernier résultat.
\end{enumerate}
\end{exo}

\begin{exo}
Soit $A= \begin{pmatrix} 1&2\\-1&4 \end{pmatrix}$, et $f$ l'endomorphisme de $\R^2$ qui a pour matrice $A$ dans la base canonique. Trouver une base dans laquelle $f$ admet pour matrice $\begin{pmatrix} 2&0\\0&3 \end{pmatrix}$. Calculer la matrice de passage.
\end{exo}
