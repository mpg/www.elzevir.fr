\begin{exo} 

Trouver les formes polaires et le rang des formes quadratiques suivantes sur $\mathbb{R}^4$
\begin{enumerate}
\item $q(x,y,z,t)=xy+y^2$ 
\item $q(x,y,z,t)=xy+zt+t^2$ 
\item $q(x,y,z,t)=x^2-y^2+z^2-t^2$
\end{enumerate}
\end{exo}

\begin{exo}
Soit q la forme quadratique sur un $\mathbb{K}$-ev associée à la forme bilinéaire symétrique f.
\begin{enumerate}
  \item Vérifier que $f(u,v)=\frac{1}{2}\left(q(u+v)-q(u)-q(v)\right)$ (forme polaire de f)
  \item Montrer que $\forall(x,y,z)\in E^3$, on a $$q(x+y)+q(y+z)+q(z+x)=q(x)+q(y)+q(z)+q(x+y+z)$$
  \end{enumerate}
\end{exo}

\begin{exo}
Soient A et B symétriques telles que $\forall$ X $\in \mathbb{R}^4$, $^tXAX=$ $^tXBX$ (*)
\begin{enumerate}
\item Soient X, Y $\in \mathbb{R}^4$, en appliquant (*) à X, Y et X+Y, montrer que $^tXAY=$ $^tXBY$
\item En remplaçant X et Y par des vecteurs de la base canonique, en déduire que A=B
  \end{enumerate}
\end{exo}
