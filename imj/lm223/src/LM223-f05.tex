\begin{exo}
~~ \newline
\begin{enumerate}
\item Trouver un changement de coordonnées qui diagonalise la forme quadratique
\[ q: \mathbb{R}^4 \rightarrow \mathbb{R} , \  
        q(\begin{pmatrix} x_1\\ x_2\\ x_3\\ x_4 \end{pmatrix})
        =x_1^2+2x_1x_2-4x_1x_4+3x_2^2+8x_2x_3+6x_3^2-2x_4^2 \ .\]
\item Déterminer la forme polaire de $q$ (=la forme bilinéaire symétrique associée à $q$).
\item Déterminer le rang de $q$.
\item Déterminer la signature de $q$.
\item Déterminer le noyau de $q$.
\end{enumerate}
\end{exo}

\begin{exo}
~~ \newline
\begin{enumerate}
  \item En utilisant une décomposition en carrés, diagonaliser la forme quadratique
  \[q: \mathbb{R}^4 \rightarrow \mathbb{R} ,\quad q(X)=4xy+4yz-2zt ,\quad X=
        \begin{pmatrix}x\\ y\\ z\\ t \end{pmatrix} \in \mathbb{R}^4\,.\]
  \item Déterminer la matrice de changement de base (ou son inverse) qui correspond à cette diagonalisation.
  \item Déterminer la signature de $q$.
  \end{enumerate}
\end{exo}

\begin{exo}
On désigne par $\alpha$ et $\beta$ deux réels tels que $\alpha^2+\beta^2=1$. Discuter suivant les valeurs de $\alpha$ et $\beta$ le rang et la signature de la forme quadratique sur $\mathbb{R}^3$ dont la matrice dans la base canonique est 
\[ \begin{pmatrix}
   {1} & {\alpha} & {0} \cr
   {\alpha} & {1} & {\beta} \cr
   {0} & {\beta} & {\alpha+\beta} 
   \end{pmatrix}       
\]       
(On utilisera le procédé d'orthogonalisation de Gauss et on distinguera les cas où $\beta=0$). Représenter graphiquement sur le cercle $\alpha^2+\beta^2=1$ les différents cas.
\end{exo}

\item Montrer que si A est symétrique, B antisymétrique, alors $f(A,B)=0$.

\begin{exo}
~~ \newline
\begin{enumerate}
\item Trouver un exemple de trois sous-espaces vectoriels $F_1$, $F_2$, $F_3$ $\in E$ tels que $F_1 \cap F_2 = F_2 \cap F_3=F_1 \cap F_3=\{0\}$ alors que la somme $F_1+F_2+F_3$ n'est pas directe.
\item Soit $\Phi$ une forme quadratique définie, on suppose que $F_1$, $F_2$, $F_3$ sont deux à deux $\Phi$-orthogonaux. Montrer que $F_1$, $F_2$ et $F_3$ sont en somme directe.
  \end{enumerate}
\end{exo}

\begin{exo}
~~ \newline
\begin{enumerate}
\item Soit $q: E \rightarrow K$ une forme quadratique sur un espace vectoriel de dimension finie. Pour tout sous-espace vectoriel $F \subset E$, montrer que l'on a ($F^\perp)^\perp=F+N(q)$.
\item Soit $q: E \rightarrow K$ une forme quadratique et $F$, $G \subset E$ des sous espaces vectoriels; montrer qu'alors $(F+G)^{\perp}=F^{\perp} \cap G^{\perp}$. Lorsque $\dim(E)<\infty$ et $q$ est non-dégénérée, montrer que l'on a $( F \cap G)^{\perp} =F^{\perp}+G^{\perp}$. 
\end{enumerate}
\end{exo}
